Customize the appearance

You can customize the appearance of this object by setting up the vars-custom map for this object. By default de values are the vars-default values.

$object-forms-vars-default: (
  /* Field */
        field-font-family: $font-face-default,
        field-font-size: $font-size-default,
        field-font-weight: normal,
        field-background: #FFF,
        field-color: #000,
        field-padding: $spacing/4 $spacing/2,
        field-margin: 0 $spacing/2 $spacing 0,
        field-border-radius: 1px,
        field-border: 1px solid rgba($color-foreground, .3),
        field-box-shadow: "inset 0 2px 8px rgba(#{$color-foreground}, .2), 0 1px 1px rgba(#{$color-background}, .5)",
        field-placeholder-color: rgba($color-foreground, .3),
  /* Field focus */
        field-focus-border: $color-info,
  /* Field error */
        field-error-border: $color-error,
  /* Label */
        label-color: rgba($color-foreground, .5)
) !global !default;

$object-forms-vars-custom: () !global !default;

$object-forms-vars: map-merge($object-forms-vars-default, $object-forms-vars-custom) !global;
