@if $object-boxes != false {

	[class*='box-'], .box {
		@include box-sizing(border-box);
		@include border-radius;
		margin-bottom: $spacing;
		padding: $spacing;
		overflow: hidden;
	}

	[class*='box-']:last-of-type {
		padding-right: $spacing;
	}

	.box {
		& {
			background: $color-box;
			color: contrast-color($color-box);
		}
		.box-lighten {
			background: $color-text-background;
		}
	}

	.box-text {
		& {
			background: $color-text-background;
			color: contrast-color($color-text-background);
		}
		& > .box-lighten {
			background: $color-box;
		}
		.box-text {
			border: 1px solid $color-box;
		}
	}

	.box-invert {
		background: $color-box-invert;
		color: contrast-color($color-box-invert);
	}

	.box-ok {
		background: $color-ok;
		color: contrast-color(darken($color-ok, 10%));
	}

	.box-ko {
		background: $color-ko;
		color: contrast-color($color-ko);
	}

	.box-info {
		background: $color-info;
		color: contrast-color($color-info);
	}

	.box-warning {
		background: $color-warning;
		color: contrast-color($color-warning);
	}

	.box-darken {
		@include background-transparent($color-text, .3);
		color: $color-text-background;
		@include box-shadow(none);
	}

	.box-lighten {
		@include background-transparent($color-text-background, .3);
		color: $color-text;
		@include box-shadow(none);
	}

	.box-main {
		& {
			background: $color-primary;
			color: contrast-color($color-primary);
		}
		a {
			color: contrast-color($color-primary);
		}
	}

	.box-aux {
		background: $color-secondary;
		color: $color-text-background;
	}
}
